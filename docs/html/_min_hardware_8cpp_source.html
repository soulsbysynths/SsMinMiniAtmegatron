293 void MinHardware::pollMidi()
294 {

296  {
297  unsigned char nextPos = (midiReadPos+1) & MIDI_BUFFER_MASK;
298  if(nextPos==midiWritePos)

300  break;
301  }
302  else

304  midiReadPos = nextPos;
305  base_->hardwareMidiReceived(midiBuffer[midiReadPos]);
306  }
307  }
308

311  base_->hardwareMidiError((unsigned char)midiError);
312  midiError = MIDIERR_NONE;
