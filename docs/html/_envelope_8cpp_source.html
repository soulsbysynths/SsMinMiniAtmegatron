116 void Envelope::refresh(unsigned char ticksPassed)
117 {
118  unsigned long inc;
119  if (update_==true){
120  switch (state_){
121  case IDLE:
122  output_ = 0;
123  update_ = false;
124  break;
125  case ATTACKING:
126  inc = (unsigned long)attack_ * ticksPassed;
127  if(inc >= (unsigned long)(MAX_VALUE - output_))
128  {
129  state_ = DECAYING;
130  output_ = MAX_VALUE;
131  }
132  else
133  {
134  output_ += inc;
135  }
136  break;
137  case DECAYING:
138  inc = (unsigned long)decay_ * ticksPassed;
139  if(inc >= (unsigned long)(output_ - sustain_))
140  {
141  state_ = SUSTAINING;
142  output_ = sustain_;
143  update_ = false;
144  }
145  else
146  {
147  output_ -= inc;
148  }
149  break;
150  case SUSTAINING:
151  output_ = sustain_;
152  update_ = false;
153  break;
154  case RELEASING:
155  inc = (unsigned long)release_ * ticksPassed;
156  if(inc >= (unsigned long)output_)
