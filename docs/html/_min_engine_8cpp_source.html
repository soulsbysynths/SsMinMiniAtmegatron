103 void MinEngine::poll(unsigned char ticksPassed)
104 {
105  unsigned char i;
106 
107  if(midi_->getClockRunning()==true)
108  {
109  masterClock_.refresh(midi_->getClockTicksPassed());
110  }
111  else
112  {
113  masterClock_.refresh(ticksPassed);
114  }
115  lfo_.refresh(masterClock_.getOutput());
116  sequencer_->refresh(masterClock_.getOutput());
117  ampEnvelope_.refresh(ticksPassed);
118  filtEnvelope_.refresh(ticksPassed);
119 
120  char lfo = lfo_.getOutput();
121  #if EXP_ENVELOPES==1
122  char env = filtEnvelope_.getExpOutput();
123  #else
124  char env = filtEnvelope_.getOutput();
125  #endif
126 
127  filter_.refresh(audio_->getSampleFreq(),lfo,env);
128  portamento_.refresh(ticksPassed);
129 
130  frequency_ = portamento_.getOutput();
131  pitch_.setInput(frequency_);
132  pitch_.refresh(lfo,env,pitchBend_);
133  audio_->setSampleFreq((unsigned long)pitch_.getOutput() * (WAVE_LENGTH >> 2));
134 
135  Wavetable working_buffer_(WAVE_LENGTH);
136  oscillator_->copyWavetable(working_buffer_);
137  pwm_->processWavetable(working_buffer_,0,lfo);
138  filter_.processWavetable(working_buffer_);
139  if(ampEnvelope_.getState()==Envelope::IDLE)
140  {
141  flanger_->clearBuffer();
142  }
143  else
144  {
145  flanger_->processWavetable(working_buffer_,0,lfo);
146  }
147  wavecrusher_.processWavetable(working_buffer_);
148  clipdistortion_.processWavetable(working_buffer_);
149  #if EXP_ENVELOPES==1
150  amplifier_.processWavetable(working_buffer_, ampEnvelope_.getExpOutput(), lfo);
151  #else
152  amplifier_.processWavetable(working_buffer_, ampEnvelope_.getOutput(), lfo);
153  #endif
154  audio_->pasteWavetable(working_buffer_);
155 }

189 void MinEngine::midiNoteOnReceived(unsigned char note, unsigned char velocity)
190 {
191  if(noteOn_[note]==false)
192  {
193  totNotesOnLast_ = totNotesOn_;
194  totNotesOn_++;
195  noteOn_[note] = true;
196  #if NOTE_PRIORITY == NP_LOW
197  if(note<noteLowest_)
198  {
199  noteLowest_ = note;
200  }
201  #elif NOTE_PRIORITY == NP_HIGH
202  if(note>noteHighest_)
203  {
204  noteHighest_ = note;
205  }
206  #elif NOTE_PRIORITY == NP_LAST
207  noteOrder_[note] = totNotesOn_;
208  noteLast_ = note;
209  #endif
210  if(sequencer_->getPattern()==0)
211  {
212  #if LEGATO==1
213  if(totNotesOn_>1)
214  {
215  portamento_.setInput(pgm_read_word(&(MIDI_FREQS[note])));
216  }
217  else
218  {
219  triggerNote(note);
220  }
221  #else
222  triggerNote(note);
223  #endif
224  }
225  else

232 void MinEngine::midiNoteOffReceived(unsigned char note)
233 {
234  if(noteOn_[note]==true)
235  {
236  totNotesOnLast_ = totNotesOn_;
237  totNotesOn_--;
238  noteOn_[note] = false;
239  #if NOTE_PRIORITY == NP_LOW
240  noteLowest_ = 127;
241  for(unsigned char i=0;i<128;++i)
242  {
243  if(noteOn_[i]==true)
244  {
245  noteLowest_ = i;
246  break;
247  }
248  }
249  #elif NOTE_PRIORITY == NP_HIGH
250  if(note==noteHighest_)
251  {
252  noteHighest_ = 0;
253  for(unsigned char i=127;i!=0;--i)
254  {
255  if(noteOn_[i]==true)
256  {
257  noteHighest_ = i;
258  break;
259  }
260  }
261  }
262  #elif NOTE_PRIORITY == NP_LAST
263  noteOrder_[note] = 0;
264  if(note==noteLast_)
265  {
266  noteLast_ = 0;
267  unsigned char last = 0;
268  for(unsigned char i=0;i<128;++i)
269  {
270  if(noteOrder_[i]>last)
271  {
272  last = noteOrder_[i];
273  noteLast_ = i;
274  }
275  }
276 
277  }
278  #endif
279  if(totNotesOn_==0 && sequencer_->getPattern()==0)
280  {
281  releaseNote();
282  }
283  else if(sequencer_->getPattern()==0)
284  {
285  #if NOTE_PRIORITY == NP_LOW
286  portamento_.setInput(pgm_read_word(&(MIDI_FREQS[noteLowest_])));
287  #elif NOTE_PRIORITY == NP_HIGH
288  portamento_.setInput(pgm_read_word(&(MIDI_FREQS[noteHighest_])));
289  #elif NOTE_PRIORITY == NP_LAST
290  portamento_.setInput(pgm_read_word(&(MIDI_FREQS[noteLast_])));
291  #endif
292  }
293  }

400 void MinEngine::patchValueChanged(unsigned char func, unsigned char newValue)
401 {
402  switch (func)
403  {
404  case FUNC_WAVE:
405  oscillator_->setBank(pgm_read_byte(&(OSC_PRESET_BANK[newValue])));
406  oscillator_->setTable(pgm_read_byte(&(OSC_PRESET_TABLE[newValue])));
407  break;
408  case FUNC_FILT:
409  filter_.setType((BiquadFilter::FiltType) newValue);
410  break;
411  case FUNC_FENV:
412  filtEnvelope_.setAttack(pgm_read_word(&(ENV_ADR_INC[pgm_read_byte(&(ENV_A_PRESET[newValue]))])));
413  filtEnvelope_.setDecay(pgm_read_word(&(ENV_ADR_INC[pgm_read_byte(&(ENV_DR_PRESET[newValue]))])));
414  filtEnvelope_.setSustain(pgm_read_word(&(ENV_S_LEVEL[pgm_read_byte(&(ENV_S_PRESET[newValue]))])));
415  filtEnvelope_.setRelease(pgm_read_word(&(ENV_ADR_INC[pgm_read_byte(&(ENV_DR_PRESET[newValue]))])));
416  break;
417  case FUNC_AENV:
418  ampEnvelope_.setAttack(pgm_read_word(&(ENV_ADR_INC[pgm_read_byte(&(ENV_A_PRESET[newValue]))])));
419  ampEnvelope_.setDecay(pgm_read_word(&(ENV_ADR_INC[pgm_read_byte(&(ENV_DR_PRESET[newValue]))])));
420  ampEnvelope_.setSustain(pgm_read_word(&(ENV_S_LEVEL[pgm_read_byte(&(ENV_S_PRESET[newValue]))])));
421  ampEnvelope_.setRelease(pgm_read_word(&(ENV_ADR_INC[pgm_read_byte(&(ENV_DR_PRESET[newValue]))])));
422  break;
423  case FUNC_LFOTYPE:
424  lfo_.setTable(newValue);
425  break;
426  case FUNC_LFOSPEED:
427  lfo_.setDivision(newValue);
428  break;
429  case FUNC_CRUSHPORTA:
430  if(newValue<8)
431  {
432  portamento_.setSpeed(pgm_read_word(&(PORTA_SPEED[0])));
433  wavecrusher_.setType(newValue<<1);
434  }
435  else
436  {
437  portamento_.setSpeed(pgm_read_word(&(PORTA_SPEED[((newValue-8)<<1)+1])));
438  wavecrusher_.setType(0);
439  }
440  break;
441  case FUNC_PATTERN:
442  sequencer_->setPattern(newValue);
443  break;
444  case FUNC_BPM:
445  setIntBpm(newValue);
446  break;
447  }
448  if(func==function_)
449  {
450  base_->engineValueChanged(func,newValue,LOW);
451  }
452 }
