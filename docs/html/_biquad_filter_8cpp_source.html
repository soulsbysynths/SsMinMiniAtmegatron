85 void BiquadFilter::refresh(unsigned long sampleFreq, char lfoOutput, char envOutput)
86 {
87  unsigned long max_fc;
88  unsigned long f;
89 
90  if(type_>BiquadFilter::FILT_OFF)
91  {
92  max_fc = sampleFreq >> 1;
93  max_fc -= sampleFreq >> 6;
94  if(!absoluteFc_)
95  {
96  f = max_fc;
97  f = f * (fcRel_ + 1) >> 8;
98  }
99  else
100  {
101  f = (unsigned long)fcAbs_;
102  }
103 
104  if(lfoAmount_>0)
105  {
106  f = f * ((unsigned int)shapeExponential(lfoOutput,lfoAmountF_,MULT_ENV_LFO) + 1) >> BS_ENV_LFO;
107  }
108  if (envAmount_>0)
109  {
110  f = f * ((unsigned int)shapeExponential(envOutput,envAmountF_,MULT_ENV_LFO) + 1) >> BS_ENV_LFO;
111  }
112  if (f>max_fc)
113  {
114  f = max_fc;
115  }
116  if(f<20)
117  {
118  f = 20;
119  }
120  }
121 
122  BiquadCalculator(type_,(float)f,(float)q_ * MULT_Q + MIN_Q, a_,M_PI/sampleFreq);

125 void BiquadFilter::BiquadCalculator(FiltType type, float fc, float q, float a, float piOverSf)
126 {
127  float w,alpha,sn,cs,sra,amcs,amps,sraalpha,apcs,k,k2p1;
128  switch (type)
129  {
130  case FILT_OFF:
131  break;
132  case FILT_LPF:
133  w = 2 * piOverSf * fc;
134  alpha = sin(w) / q * 0.5;
135  cs = cos(w);
136  b0_ = (1 - cs) / 2;
137  b1_ = 1 - cs;
138  b2_ = b0_;
139  a0_ = 1 + alpha;
140  a1_ = -2 * cs;
141  a2_ = 1 - alpha;
142  break;
143  case FILT_HPF:
144  w = 2 * piOverSf * fc;
145  alpha = sin(w) / q * 0.5;
146  cs = cos(w);
147  b0_ = (1 + cs) / 2;
148  b1_ = -(1 + cs);
149  b2_ = b0_;
150  a0_ = 1 + alpha;
151  a1_ = -2 * cs;
152  a2_ = 1 - alpha;
153  break;
154  case FILT_BPF:
155  w = 2 * piOverSf * fc;
156  alpha = sin(w) / q * 0.5;
157  sn = sin(w);
158  cs = cos(w);
159  b0_ = sn/2;
160  b1_ = 0;
161  b2_ = -b0_;
162  a0_ = 1 + alpha;
163  a1_ = -2*cs;
164  a2_ = 1 - alpha;
165  break;
166  case FILT_NOTCH:
167  w = 2 * piOverSf * fc;
168  alpha = sin(w) / q * 0.5;
169  cs = cos(w);
170  b0_ = 1;
171  b1_ = -2*cs;
172  b2_ = 1;
173  a0_ = 1 + alpha;
174  a1_ = b1_;
175  a2_ = 1 - alpha;
176  break;
177  case FILT_PEAK10:
178  case FILT_PEAK30:
179  case FILT_PEAK100:
180  w = 2 * piOverSf * fc;
181  alpha = sin(w) / q * 0.5;
182  cs = cos(w);
183  b0_ = 1 + alpha*a;
184  b1_ = -2*cs;
185  b2_ = 1 - alpha*a;
186  a0_ = 1 + alpha/a;
187  a1_ = b1_;
188  a2_ = 1 - alpha/a;
189  break;
190  case FILT_LS10:
191  case FILT_LS30:
192  w = 2 * piOverSf * fc;
193  cs = cos(w);
194  sra = sqrt(a);
195  alpha = sin(w) / q * 0.5;
196  amcs = (a-1)*cs;
197  apcs = (a+1)*cs;
198  sraalpha = 2*sra*alpha;
199  b0_ = a*( (a+1) - amcs + sraalpha );
200  b1_ = 2*a*( (a-1) - apcs );
201  b2_ = a*( (a+1) - amcs - sraalpha );
202  a0_ = (a+1) + amcs + sraalpha ;
203  a1_ = -2*( (a-1) + apcs );
204  a2_ = (a+1) + amcs - sraalpha ;
205  break;
206  case FILT_HS10:
207  case FILT_HS30:
208  w = 2 * piOverSf * fc;
209  cs = cos(w);
210  sra = sqrt(a);
211  alpha = sin(w) / q * 0.5;
212  amcs = (a-1)*cs;
213  apcs = (a+1)*cs;
214  sraalpha = 2*sra*alpha;
215  b0_ = a*( (a+1) + amcs + sraalpha );
216  b1_ = -2*a*( (a-1) + apcs );
217  b2_ = a*( (a+1) + amcs - sraalpha );
218  a0_ = (a+1) - amcs + sraalpha ;
219  a1_ = 2*( (a-1) - apcs );
220  a2_ = (a+1) - amcs - sraalpha ;
221  break;
222  case FILT_BUTLPF:
223  k = tan(fc * piOverSf);
224  b2_ = k * k;
225  b0_ = b2_;
226  b1_ = 2 * b0_;
227  a0_ = b0_ + (M_SQRT2 * k) + 1;
228  a1_ = 2 * (b0_ - 1);
229  a2_ = b0_ - (M_SQRT2 * k) + 1;
230  break;
231  case FILT_BUTHPF:
232  k = tan(fc * piOverSf);
233  k2p1 = k * k + 1;
234  b0_ = 1;
235  b2_ = b0_;
236  b1_ = -2;
237  a0_ = k2p1 + (M_SQRT2 * k);
238  a1_ = 2 * (k2p1 - 2);
239  a2_ = k2p1 - (M_SQRT2 * k);
240  break;
241  case FILT_BESLPF:
242  w = tan(piOverSf * fc);
243  b2_ = 3 * w * w;
244  b0_ = b2_;
245  b1_ = 2 * b0_;
246  a0_ = 1 + 3 * w + b0_;
247  a1_ = -2 + b1_;
248  a2_ = 1 - 3 * w + b0_;
249  break;
250  case FILT_BESHPF:
251  w = tan(piOverSf * fc);
252  float w2 = w * w;
253  b2_ = 3;
254  b0_ = b2_;
255  b1_ = -6;
256  a0_ = w2 + 3 * w + 3;
257  a1_ = 2 * w2 - 6;
258  a2_ = w2 - 3 * w + 3;
259  break;
260  }
261  b0_ /= a0_;
262  b1_ /= a0_;
263  b2_ /= a0_;
264  a1_ /= a0_;
265  a2_ /= a0_;
266 }

320 float BiquadFilter::BiquadProcess(float bi0)
321 {
322  float bo0;
323  bo0 = b0_ * bi0 + b1_ * bi1_ + b2_ * bi2_ - a1_ * bo1_ - a2_ * bo2_;
324  bi2_ = bi1_;
325  bi1_ = bi0;
326  bo2_ = bo1_;
327  bo1_ = bo0;
328  return bo0;
