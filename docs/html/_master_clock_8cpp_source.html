29 void MasterClock::refresh(unsigned char ticksPassed)
30 {
31  ticks_ += ticksPassed;
32  if(ticks_>=ticksPerCycle_)
33  {
34  ticks_ -= ticksPerCycle_;
35  }
36  unsigned char bs = bsScaleConstrain(ticks_);
37  output_ = (ticks_ << (16-bs)) / (ticksPerCycle_ >> bs);
38 }
