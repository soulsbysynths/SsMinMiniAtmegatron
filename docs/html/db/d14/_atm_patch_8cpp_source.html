108 void AtmPatch::readPatch(unsigned char patchNum)
109 {
110  unsigned int startAddr = (unsigned int)PATCH_SIZE * patchNum;
111  unsigned int addr = 0;
112  unsigned char i,j;
113  unsigned char data[PATCH_SIZE];
114  readMemory((void*)data,(const void*)startAddr, sizeof(data));
115  for(i=0;i<FUNC_CS;++i)
116  {
117  setFunctionValue(i<<1,uncompressFourBit(data[addr],false));
118  setFunctionValue((i<<1)+1,uncompressFourBit(data[addr],true));
119  addr++;
120  }
121  for(i=0;i<BANKS;++i)
122  {
123  for(j=0;j<CTRLS;++j)
124  {
125  setCtrlValue(i,j,data[addr]);
126  addr++;
127  }
128  }
129  for(i=0;i<OPT_CS;++i)
130  {
131  for(j=0;j<OPTS;++j)
132  {
133  setOptionValue((i*OPTS)+j,(bool)bitRead(data[addr],j));
134  }
135  addr++;
136  }
