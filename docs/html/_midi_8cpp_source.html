45 void Midi::read(unsigned char inByte)
46 {
47  static unsigned char noteByte=0;
48  static unsigned char velByte=0;
49  static unsigned char statusBuffer=0;
50  static bool firstByte;
51  static unsigned char sysexRxMnftr = 255;
52  static unsigned char sysexRxProdid = 255;
53 
54  if (inByte>=CLOCK)
55  {
56  switch (inByte)
57  {
58  case CLOCK:
59  clockTicks_++;
60  break;
61  case START:
62  case CONTINUE: //currently implemented same as start
63  clockTicks_ = 0;
64  clockTicksLast_ = 0;
65  clockRunning_ = true;
66  base_->midiClockStartReceived();
67  break;
68  case STOP:
69  clockRunning_ = false;
70  base_->midiClockStopReceived();
71  break;
72  }
73  }
74  else if (inByte>=SYSEX_START)
75  {
76  statusBuffer = 0;
77  if (inByte==SYSEX_START)
78  {
79  sysexRx_ = true;
80  sysexIndex_ = 0;
81  sysexRxMnftr = 255;
82  sysexRxProdid = 255;
83  if(sysexEnable_==true)
84  {
85  base_->midiSysexStartReceived();
86  }
87 
88  }
89  if (inByte==SYSEX_STOP && sysexRx_==true)
90  {
91  sysexRx_ = false;
92  if(sysexEnable_==true)
93  {
94  base_->midiSysexStopReceived();
95  }
96  }
97  }
98  else if (inByte>=NOTE_OFF)
99  {
100  statusBuffer = inByte;
101  firstByte = true;
102  noteByte = 0;
103  velByte = 0;
104  sysexRx_ = false;
105  }
106  else if (sysexRx_==true)
107  {
108  sysexIndex_++;
109  if(sysexIndex_==1)
110  {
111  sysexRxMnftr = inByte;
112  }
113  else if(sysexIndex_==2)
114  {
115  sysexRxProdid = inByte;
116  }
117  else if (sysexRxMnftr==SYSEX_MANUFACTURER && sysexRxProdid==sysexProductId_)
118  {
119  if(sysexEnable_==true)
120  {
121  base_->midiSysexDataReceived(sysexIndex_-3,inByte);
122  }
123  }
124  }
125  else if (statusBuffer!=0)
126  {
127  if (firstByte == true)
128  {
129  noteByte = inByte;
130  firstByte = false;
131  if (progChangeEnable_==true && statusBuffer==(PROGRAM_CHANGE | channel_))
132  {
133  base_->midiProgramChangeReceived(noteByte);
134  }
135  }
136  else
137  {
138  velByte = inByte;
139  if (statusBuffer == (NOTE_ON | channel_) && velByte > 0)
140  {
141  if(noteOn_[noteByte]==false)
142  {
143  noteOn_[noteByte] = true;
144  totNotesOn_++;
145  base_->midiNoteOnReceived(noteByte,velByte);
146  }
147 
148  }
149  else if (statusBuffer == (NOTE_OFF | channel_) || (statusBuffer == (NOTE_ON | channel_) && velByte == 0))
150  {
151  if(noteOn_[noteByte]==true)
152  {
153  noteOn_[noteByte] = false;
154  totNotesOn_--;
155  base_->midiNoteOffReceived(noteByte);
156  }
157  }
158  else if (statusBuffer == (CONTROL_CHANGE | channel_))
159  {
160  base_->midiControlChangeReceived(noteByte,velByte);
161  }
162  else if (statusBuffer == (PITCH_BEND | channel_))
163  {
164  if(velByte==0)
165  {
166  base_->midiPitchBendReceived(-127);
167  }
168  else if (velByte==127)
169  {
170  base_->midiPitchBendReceived(127);
171  }
172  else
173  {
174  base_->midiPitchBendReceived((velByte-64)<<1);
175  }
176  }
177 
178  noteByte = 0;
179  velByte = 0;
180  firstByte = true;
181  }
182  }
183 }
